# 高精度运算

高精度运算主要用来处理范围大于正常基本数据类型的数据，比如实现`1e7`位数字的加减乘除计算。

### 加法



```c++
#include <iostream>
#include <vector>
using namespace std;

// 这里的数据采取高位数字放在数组的低位，如 12345在数组中存储为[5,4,3,2,1],方便进位计算
vector<int> add(vector<int> &A, vector<int> &B){
	vector<int> C;
    
    int t = 0;
    for(int i = 0; i < A.size() || i < B.size(); i++){
        if (i < A.size()) t += A[i];
        if (i < B.size()) t += B[i];
        C.push_back(t%10);
        t /= 10;
    }
    if (t) C.push_back(1);
    return C;
}


int main(){
    string S_A, S_B;
    
    cin >> S_A >> S_B;
    
    vector<int> A, B;
    
    /*数据存储进行处理*/
    for(int i = S_A.size() - 1; i >= 0; i--){
        A.push_back(S_A[i] - '0');
    }
    for(int i = S_B.size()-1; i >= 0; i--){
        B.push_back(S_B[i] - '0');
    }
    
    auto C = add(A, B);
     
    for(int i = C.size()-1; i >= 0; --i){
        cout << C[i];
    }
}
```

### 减法

减法中我们规定A - B中A一定要比B要大，这样便于整理一个通用的减法模板，但若遇到B比A大，那么将计算式子变为等价形式 A-B = - (B - A)，这样只需要计算后面的部分并在输出前加上负号即可。

```c++
#include <iostream>
#include <vector>
using namespace std;

vector<int> sub(vector<int> &A, vector<int> &B){
    vector<int> C;
    // 输入的A是要比B大的
    for(int i = 0, t = 0; i < A.size();i++){
        t = A[i]-t;
        if (i < B.size()) t -= B[i];	//此时i为一位的计算结果
        C.push_back((t + 10)%10);
        if (t < 0) t = 1;	//上次操作借位了
        else t = 0;			//没借位
    }
    
    //消除前导0,但是最低要保留一个零（A与B相同时）
    while(C.size() > 1 && C.back() == 0) C.pop_back();
    return C;
    
}

//判断A是否比B大
bool cmp(vector<int> &A, vector<int> &B){
    //位数不同，直接比较位数大小
    if (A.size() != B.size()) return A.size() > B.size();
    
    //位数相同，从最高位开始比较
    for(int i = A.size() - 1; i >= 0; i--){
        if (A[i] != B[i]) return A[i] > B[i];
    }
    
    // 两数相同，是可以直接进入sub进行计算的
    return true;
}

int main(){
    string a,b;
    cin >> a >> b;
    
    vector<int> A, B, C;
    for(int i = a.size()-1; i >= 0; i--) A.push_back(a[i] - '0');
    for(int i = b.size()-1; i >= 0; i--) B.push_back(b[i] - '0');
    
    if (cmp(A, B)){
        C = sub(A, B);
    }
    else{
        C = sub(B,A);
        cout << "-";
    }
    for(int i = C.size()-1; i >= 0; i--) cout << C[i];
}
```
