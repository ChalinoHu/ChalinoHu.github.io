#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <arpa/inet.h>
#include <pthread.h>
#include <string.h>

// 创建结构体存储线程需要用到的数据
struct sockInfo{
    int client_num;	// 客户端的ID
    int fd;     // 通信的文件描述符
    pthread_t tid;  // 线程号
    struct sockaddr_in addr;
}sockinfos[128];	// 设定最大接收128个客户端


void* working(void * arg){
	// 参数解包
    struct sockInfo *pinfo = (struct sockInfo *)arg;

    // 打印出客户端的信息
    unsigned short port = ntohs(pinfo->addr.sin_port);	// 端口
    char buf[16] = {0};
    inet_ntop(AF_INET, &pinfo->addr.sin_addr, buf, sizeof buf);	// 网络字节序转点分十进制
    printf("client %d ip is %s, port is %d\n", pinfo->client_num, buf, port);
    // 子线程和客户端通信
    char recvBuf[1024] = {0};
    while(1){
        
        // 读取客户端发来的数据
        memset(recvBuf, 0, sizeof recvBuf);
        int len = read(pinfo->fd, recvBuf, sizeof recvBuf);
        if (len == -1){
            perror("read");
            exit(-1);
        }else if (len > 0){
            printf("Receive client %d massage: %s\n", pinfo->client_num, recvBuf);
        }else {	// len == 0
            printf("client closed...");
            break;
        }

        // 给客户端发送的消息

        write(pinfo->fd, recvBuf, strlen(recvBuf) + 1);
    }
    return NULL;
}

int main(){

    // 创建socket
    int lfd = socket(AF_INET, SOCK_STREAM, 0);
    if (lfd == -1){
        perror("socket");
        exit(-1);
    }

    // 绑定
    struct sockaddr_in saddr;
    saddr.sin_addr.s_addr = INADDR_ANY;
    saddr.sin_family = AF_INET;
    saddr.sin_port = htons(9999);
    int ret = bind(lfd, (struct sockaddr *)&saddr, sizeof saddr);
    if (ret == -1){
        perror("bind");
        exit(-1);
    }

    // 监听
    ret = listen(lfd, 128); // 128 监听队列的长度
    if (ret == -1){
        perror("listen");
        exit(-1);
    }

    // 初始化传入线程的数据
    int max = sizeof sockinfos / sizeof sockinfos[0];
    for (int i = 0; i < max; i++){
        bzero(&sockinfos[i], sizeof sockinfos[i]);
        sockinfos[i].fd = -1;
        sockinfos[i].tid = -1;
        sockinfos[i].client_num = -1;
    }

    // 循环等待客户端连接，一旦一个客户端连接进来， 就创建一个子线程跟客户端通信
    int client_num = 0;
    while (1){

        struct sockaddr_in cliaddr;
        socklen_t len = sizeof cliaddr;
        int cfd = accept(lfd, (struct sockaddr *)&cliaddr, &len);
        if (cfd == -1){
            perror("accept");
            exit(-1);
        }
        client_num++;
        struct sockInfo * pinfo;
        
        for (int i = 0; i < max; i++){
            // 从这个数组找到一个可以用的sockinfo元素
            if (sockinfos[i].fd == -1){
                pinfo = &sockinfos[i];
                break;
            }
            if (i == max - 1){
                sleep(1);
                i = 0;  // 从头开始找
            }
        }
        pinfo->client_num = client_num;
        pinfo->fd = cfd;
        memcpy(&pinfo->addr, &cliaddr ,sizeof cliaddr);
        // 创建子线程

        pthread_create(&pinfo->tid, NULL, working, pinfo);
        pthread_detach(pinfo->tid);
    }

    return 0;
}
