```c
#include <stdio.h>
#include <unistd.h>
#include <arpa/inet.h>
#include <stdlib.h>
#include <string.h>
#include <sys/types.h>
#include <signal.h>
#include <wait.h>
#include <errno.h>

void recycleChild(int arg){
    while (1){
        int ret = waitpid(-1, NULL, WNOHANG);
        if (ret == -1){
            // 所有的子进程都回收了
            break;
        }else if (ret == 0){
            // 还有子进程活着
            break;
        }else if (ret > 0){
            // 被回收了
            printf("子进程 %d 被回收了\n", ret);
        }
    }

}


int main(){

    // 注册信号捕捉，父进程回收子进程的资源
    struct sigaction act;
    act.sa_flags = 0;
    sigemptyset(&act.sa_mask);
    act.sa_handler = recycleChild;
    sigaction(SIGCHLD, &act, NULL);

    // 创建socket
    int lfd = socket(AF_INET, SOCK_STREAM, 0);
    if (lfd == -1){
        perror("socket");
        exit(-1);
    }

    // 绑定
    struct sockaddr_in saddr;
    saddr.sin_family = AF_INET;
    // inet_pton(AF_INET, "0.0.0.0", &saddr.sin_addr.s_addr);
    saddr.sin_addr.s_addr = INADDR_ANY;
    saddr.sin_port = htons(9999);

    int ret = bind(lfd, (struct sockaddr *)&saddr, sizeof saddr);
    if (ret == -1){
        perror("bind");
        exit(-1);
    }

    // 监听
    ret = listen(lfd, 128);
    if (ret == -1){
        perror("listen");
        exit(-1);
    }

    int num = 0;
    // 接收连接
    // 循环等待客户端连接
    while (1){

        // 接受连接
        struct sockaddr_in cliaddr;
        int len = sizeof cliaddr;
        int cfd = accept(lfd, (struct sockaddr*) &cliaddr, &len);
        if (cfd == -1){
            if (errno == EINTR){    // accept在阻塞状态激进入信号处理函数后会直接越过阻塞状态接着向下执行，因此返回-1
                continue;           // 判断如果是因为信号的原因而返回-1，则不让程序终止
            }
            perror("accept");
            exit(-1);
        }

        // 每一个客户端连接进来，创建一个子进程去和客户端通信
        num++;
        if (fork() == 0){   // 子进程
            // 获取客户端的信息
            char client_ip[16] = "";    // ip地址
            inet_ntop(AF_INET, &cliaddr.sin_addr.s_addr, client_ip, sizeof client_ip);
            unsigned short client_port = ntohl(cliaddr.sin_port);

            printf("The client %d connected, ip is %s, port is %d\n", num, client_ip, client_port);

            // 开始通信
            // 先接受客户端发来的信息
            char recvBuf[1024] = {0};
            while (1){
                memset(recvBuf, 0, sizeof recvBuf);
                int len = read(cfd, recvBuf, sizeof recvBuf);
                if (len == -1){
                    perror("read");
                    exit(-1);
                }else if (len > 0){
                    printf("recv client %d data: %s\n", num, recvBuf);
                }else {
                    printf("client closeed...\n");
                    break;
                }

                write(cfd, recvBuf, strlen(recvBuf) + 1);
            }

            close(cfd);
        }
        
    }

    close(lfd);
    return 0;
}
```

客户端：

```c
#include <arpa/inet.h>
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>

int main(){

    int fd = socket(PF_INET, SOCK_STREAM, 0);
    if (fd == -1){
        perror("socket");
        exit(-1);
    }

    struct sockaddr_in cliaddr;
    cliaddr.sin_family = AF_INET;
    inet_pton(AF_INET, "192.168.198.128", &cliaddr.sin_addr.s_addr);
    cliaddr.sin_port = htons(9999);

    int ret =connect(fd, (struct sockaddr*) &cliaddr, sizeof cliaddr);
    if (ret == -1){
        perror("connect");
        exit(-1);
    }

    // 通信
    char buf[1024] = {0};
    while (1){
        
        // 从终端读数据
        memset(buf, 0, sizeof buf);
        read(STDIN_FILENO, buf, sizeof buf);

        // 向服务端发送数据
        write(fd, buf, strlen(buf) + 1);
        

        // 从服务端读数据
        memset(buf, 0, sizeof buf);
        ret = read(fd, buf, sizeof buf);
        if (ret == -1){
            perror("read");
            exit(-1);
        }else if (ret > 0){
            printf("recv server data: %s\n", buf);
        }else {
            printf("server closed..\n");
            break;
        }
    }

    close(fd);

    return 0;
}
```
